async function loadGroups() { const accountId = document.getElementById('accountSelect').value; const container = document.getElementById('groupContainer'); if(!accountId) { alert('Pilih akun dulu bos!'); return; } container.innerHTML = 'Sedang menginspeksi kode Facebook, mohon tunggu...'; try { const res = await fetch('/api/fetch-groups', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ accountId }) }); const data = await res.json(); if(data.success && data.debugHtml) { // Kita tampilkan 1000 karakter pertama isi HTML Facebook di kotak grup buat kita periksa // Dan kita cetak full kodenya di Console Browser (F12) console.log("HTML LENGKAP FACEBOOK:", data.debugHtml); // Cari tahu apakah ada indikasi terlempar ke halaman login/checkpoint let statusLog = "Koneksi Berhasil. "; if(data.debugHtml.includes("login.php")) statusLog = "MENINGGAL: Terlempar ke Halaman Login! "; if(data.debugHtml.includes("checkpoint")) statusLog = "MENINGGAL: Akun Terkena Checkpoint/Sesi Habis! "; container.innerHTML = ` Status Analisis: ${statusLog}


Isi kodingan lengkap sudah dikirim ke Console F12 browser kamu. `; } else { container.innerHTML = `Gagal: ${data.error}`; } } catch (err) { container.innerHTML = 'Terjadi kesalahan koneksi server.'; } }
